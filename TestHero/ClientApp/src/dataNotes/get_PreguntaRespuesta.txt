DELIMITER //
DROP PROCEDURE IF EXISTS get_PreguntaRespuesta;
CREATE PROCEDURE get_PreguntaRespuesta(IN idE int)
BEGIN
	select pregunta.pregunta, respuesta.respuesta, respuesta.esCorrecta
    from pregunta inner join respuesta on
     pregunta.idPregunta = respuesta.idPregunta and pregunta.idExamen = idE;
END 
// DELIMITER ;


call  get_PreguntaRespuesta(1)
