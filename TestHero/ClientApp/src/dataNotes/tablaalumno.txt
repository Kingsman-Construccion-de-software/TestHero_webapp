DELIMITER //
DROP PROCEDURE IF EXISTS get_tablaalumno;
CREATE PROCEDURE  get_tablaalumno(IN idG int)
BEGIN
	SELECT alumno.nombres, alumno.apellidos, grupo.nombre as Grupo, alumnoexamen.calificacion 
    from alumno inner join grupo on 
    alumno.idGrupo = grupo.idGrupo 
    inner join alumnoexamen on
	alumno.idAlumno  = alumnoexamen.idAlumno and alumnoexamen.idExamen = idG;

END 
// DELIMITER ;


call   get_tablaalumno(2)
