DELIMITER //
DROP PROCEDURE IF EXISTS insert_alumnoexamen;
CREATE PROCEDURE  insert_alumnoexamen(IN idA int,IN idE int,IN cali int,IN pun int, IN fRealizacion datetime)
BEGIN
	INSERT INTO  alumnoexamen(idAlumno,idExamen,calificacion,puntos,fechaRealizacion) 
    values(idA,idE,cali,pun,fRealizacion);
END //
DELIMITER ;

call insert_alumnoexamen(7,1,5,100,now());
